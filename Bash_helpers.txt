#Simple process parallelism

#!/bin/bash
for n in $(cat xx.cfg)
do
 DIRECTORY=$(echo $n | cut -d "," -f 1)
 FILES=$(echo $n | cut -d "," -f 2)
 RESULTS=$(echo $n | cut -d "," -f 3)
 CMD="find "${DIRECTORY}" -name "${FILES}" > "${RESULTS}
 eval ${CMD} &
done
wait
echo "Script execution completed"
